[create] tbl_student을 만드는 SQL문을 작성
create table tbl_student (
	id int primary key,
    name varchar(45),
    addr varchar(45),
    phone varchar(45)
);

[alter] 열을 추가하는 SQL문을 작성 (gender varchar(1) null) 
alter table tbl_student add column gender varchar(1) null;

[drop] phone열을 제거 하는 SQL문을 작성
alter table tbl_student drop column phone;

[insert] SQL문으로 다음 내용이 삽입될 수 있도록 하는 INSERT 문을 작성
insert into tbl_student values
    (1, '홍길동', '대구', 'M'),
    (2, '기명이', '울산', 'M'),
    (3, '김미녀', '인천', 'W');

[update] SQL문으로 다음 내용이 수정될 수 있도록 하는 UPDATE 문을 작성
update tbl_student set name = '남길동' where id = 1;
update tbl_student set addr = '창원' where id = 3;

[delete] SQL문으로 다음 내용이 삭제될 수 있도록 하는 DELETE 문을 작성
delete from tbl_student where id = 2;
