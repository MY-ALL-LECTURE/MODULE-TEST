3. DDL / DML 사용하기

3-1.테이블 생성 
create table tbl_student(
    -> id int primary key,
    -> name varchar(45) not null,
    -> addr varchar(45) not null,
    -> phone varchar(45) not null);

3-2 열 추가 

alter table tbl_student add column gender varchar(1) null after phone;

3-3 열 제거 

alter table tbl_student drop phone;

3-4 내용 삽입

insert into testdb.tbl_student(id, name, addr, gender) values('1','홍길동','대구','M');

insert into testdb.tbl_student(id, name, addr, gender) values('2','기명이','울산','M');

insert into testdb.tbl_student(id, name, addr, gender) values('3','김미녀','인천','W');

3-5 내용 수정 

update tbl_student set addr='창원' where id ='3';

3-6 내용 삭제

delete from tbl_student where  id='2';
