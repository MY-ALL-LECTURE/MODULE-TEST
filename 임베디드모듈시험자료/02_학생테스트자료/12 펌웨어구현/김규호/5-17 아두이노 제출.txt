const unsigned int led[5] = {2,3,4,5,6}; //음수값을 지정하지 않는 크기5의 int형 배열 생성
void setup() // // 아두이노 보드가 실행되면서 제일 먼저 읽는 부분, 사용할 함수들을 선언하는 공간
{
  Serial.begin(115200); // 아두이노가 시리얼 통신을 하도록 만드는 함수 초당 몇 비트의 속도로 통신 할 지 설정
}

void loop() // setup() method 에서 선언한 변수를 어떻게 사용할 것 인지 정하는 곳, 특정 조건식을 미설정 시 무한 반복된다.
{
  
  for(int i=0; i<5; i++){ // for문 시작 int형 변수 i 생성, i<5까지 반복, 한번의 사이클 돌때마다 i값 1씩 증가
    digitalWrite(led[i],HIGH); // led[i] 디지털 핀의 내부 풀업 저항을 활성화, 출력 100
    delay(500);  // 프로그램의 진행을 0.5초 동안 일시 정지한다 ms 단위로 1000ms = 1s 이다.
    digitalWrite(led[i],LOW);  // led[i] 디지털 핀의 내부 풀업 저항을 비 활성화, 출력 0 
    
  }
}
